Script para Google Apps Script

Copiá este código y pegalo en tu Google Apps Script:

// ===== PRODE MUNDIAL 2026 - Google Apps Script ===== // Pegá este código en Extensions > Apps Script de tu Google Sheet function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); if (data.action === 'guardar_prono') { let tab = sheet.getSheetByName(data.jugador) || sheet.insertSheet(data.jugador); tab.clearContents(); tab.getRange(1,1,1,4).setValues([['Partido','Local','Visitante','Guardado']]); const rows = data.partidos.map(p => [p.id, p.local||'', p.visitante||'', new Date().toISOString()]); if (rows.length) tab.getRange(2,1,rows.length,4).setValues(rows); // Bonus if (data.bonus) { let bTab = sheet.getSheetByName('BONUS') || sheet.insertSheet('BONUS'); const bRow = bTab.createTextFinder(data.jugador).findNext(); const bData = [data.jugador, data.bonus.campeon||'', data.bonus.subcampeon||'', data.bonus.goleador||'', data.bonus.sorpresa||'']; if (bRow) { bTab.getRange(bRow.getRow(),1,1,5).setValues([bData]); } else { bTab.appendRow(bData); } } return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } if (data.action === 'guardar_resultados') { let tab = sheet.getSheetByName('RESULTADOS') || sheet.insertSheet('RESULTADOS'); tab.clearContents(); tab.getRange(1,1,1,3).setValues([['Partido','Local','Visitante']]); const rows = data.partidos.map(p => [p.id, p.local !== '' ? p.local : '', p.visitante !== '' ? p.visitante : '']); if (rows.length) tab.getRange(2,1,rows.length,3).setValues(rows); return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({error:'accion desconocida'})).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet(); const action = e.parameter.action; if (action === 'tabla') { const res = getResultados(sheet); const jugadores = sheet.getSheets() .filter(s => !['RESULTADOS','BONUS','Sheet1'].includes(s.getName())) .map(s => calcularPuntosJugador(s, res, sheet)); jugadores.sort((a,b) => b.pts - a.pts); return ContentService.createTextOutput(JSON.stringify(jugadores)).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({error:'sin accion'})).setMimeType(ContentService.MimeType.JSON); } function getResultados(sheet) { const tab = sheet.getSheetByName('RESULTADOS'); if (!tab) return {}; const data = tab.getDataRange().getValues().slice(1); const res = {}; data.forEach(r => { if(r[0]) res[r[0]] = {local: r[1], visitante: r[2]}; }); return res; } function calcularPuntosJugador(tab, resultados, sheet) { const nombre = tab.getName(); const pronos = tab.getDataRange().getValues().slice(1); let pts = 0, exactos = 0, ganadores = 0; pronos.forEach(p => { const id = p[0]; const pLocal = p[1]; const pVis = p[2]; const real = resultados[id]; if (!real || real.local === '' || real.visitante === '') return; if (pLocal === real.local && pVis === real.visitante) { pts += 5; exactos++; } else if (Math.sign(pLocal - pVis) === Math.sign(real.local - real.visitante)) { pts += 2; ganadores++; } }); // Bonus let bonus = 0; const bTab = sheet.getSheetByName('BONUS'); if (bTab) { const bData = bTab.getDataRange().getValues().slice(1); const bRow = bData.find(r => r[0] === nombre); if (bRow) { const bActual = resultados['bonus'] || {}; if (bRow[1] && bRow[1] === bActual.campeon) bonus += 10; if (bRow[2] && bRow[2] === bActual.subcampeon) bonus += 5; if (bRow[3] && bRow[3] === bActual.goleador) bonus += 8; } } return { nombre, pts: pts + bonus, exactos, ganadores, bonus }; }
